6 Source code

num_dict = {}
def fact(num):
    if num == 1:
	return 1
    if (num - 1) in num_dict:
	return num * fact(num - 1)
    else:
	num_dict[num-1] = fact(num - 1)
	return num * fact(num - 1)
print(fact(4))
